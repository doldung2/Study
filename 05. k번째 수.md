05. k번째 수
### java
```java
import java.util.*;
class Solution {
    public int[] solution(int[] array, int[][] commands) {
        
        int cmndCnt = commands.length;
        int cmndLen = 3;
        
        int[] answer = new int[cmndCnt];
        
        for(int p=0;p<cmndCnt;p++){
            
            int i = commands[p][0];
            int j = commands[p][1];
            int k = commands[p][2];
            int o = 0;
            
            int[] rsltArr = new int[j-i+1];
            for(int x=0;x<j;x++){
                if(x>=i-1){
                    rsltArr[o] = array[x];
                    o = o + 1;
                }
            }
            Arrays.sort(rsltArr);
            answer[p] = rsltArr[k-1];
        }
        
        return answer;
    }
}
```
