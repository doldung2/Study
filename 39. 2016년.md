class Solution {
    public String solution(int a, int b) {
        
        String[] week = {"SUN","MON","TUE","WED","THU","FRI","SAT"};
        int days = 0;
        
        for(int i=1;i<a;i++){
            if(i==4||i==6||i==9||i==11){
                days += 30;
            }else if(i==2){
                days += 29;
            }else{
                days += 31;
            }
        }
        days += b+4;        
        String answer = week[days%7];
        
        return answer;
    }
}
